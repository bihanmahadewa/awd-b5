# Shell Scripting

## Make a shell file
    $ nano first.shell
    
## Inside the shell file add the following code
    $ #!/bin/bash/
    $ echo "HEllo SHELL SCRIPTING"
    
 Save the File and Exit
 
 ## Run the shell 
    $ bash first.sh
 
 # Second Example
 
 ## Make a new shell file
    $ nano second.shell
 
 ## Inside the shell file add the following code
    $ #!/bin/bash
    $ echo "Enter Your Name ?"
    $ read name
    $ echo "Your Name is" $name
